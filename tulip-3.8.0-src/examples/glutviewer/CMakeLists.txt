cmake_minimum_required(VERSION 2.6)

FIND_PACKAGE( TULIP3 REQUIRED )
FIND_PACKAGE( OpenGL REQUIRED)
FIND_PACKAGE(GLUT REQUIRED)

INCLUDE_DIRECTORIES( ${TULIP_INCLUDE_DIR} ${OPENGL_INCLUDE_DIR} ${GLUT_INCLUDE_DIRS})

SET(LIB_SRCS glutviewer.cpp)

ADD_EXECUTABLE(renderer ${LIB_SRCS} )
TARGET_LINK_LIBRARIES(renderer ${GLUT_LIBRARY})
TARGET_LINK_LIBRARIES(renderer ${OPENGL_glu_LIBRARY})
TARGET_LINK_LIBRARIES(renderer ${OPENGL_gl_LIBRARY})
TARGET_LINK_LIBRARIES(renderer ${TULIP_LIBS})
